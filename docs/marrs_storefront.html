            p1 = Proj(dataset.crs.wkt) # our current crs
            p2 = Proj("+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs ") # the crs we want our raster to be in
            raster_utm_points = dataset.transform * (0, 0), dataset.transform * (dataset.width, 0), dataset.transform * (dataset.width, dataset.height), dataset.transform * (0, dataset.height)
            lon, lat = transform(p1, p2, np.array(raster_utm_points)[:,0],np.array(raster_utm_points)[:,1])
            raster_gps_points = list(zip(lat, lon)) # zip up the points so they're in the form [(lat, long), ...]
            gps_point_sets.append(raster_gps_points)

            basename = os.path.basename(fp)
            date = basename[:6]

            flight_num = basename.split("_")[0][-2:]
            sensor_type = basename.split("_")[0][-4:-2]
            location = basename.split("_")[0][6:-4]
            if not flight_num.isdigit():
                flight_num = "01"
                sensor_type = basename.split("_")[0][-2:]
                location = basename.split("_")[0][6:-2]

            metadata = {
                "crs" : dataset.crs.wkt,
                "location" : location,
                "date" : date,
                "flight_num" : flight_num,
                "sensor_type" : sensor_type,
                "filename" : fp,
                "shape" : dataset.shape,
                "size" : dataset.shape[0] * dataset.shape[1],
                "bands" : dataset.count,

            }

            metadata_list.append(metadata)
    except:
        continue
